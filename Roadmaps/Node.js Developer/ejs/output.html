const ejs = require('ejs'); const fs = require('fs'); // Data to be passed to the template const data = { name: 'blue', people: [ 'geddy', 'neil', 'alex' ] }; // Render the template ejs.renderFile('./views/1.ejs', data, function (err, renderedHTML) { if (err) { console.error(err); } else { // Log the rendered HTML console.log(renderedHTML); // Optionally, save the rendered HTML to a file fs.writeFileSync('./views/rendered.html', renderedHTML, 'utf8'); } });
